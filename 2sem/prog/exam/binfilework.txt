	size_t fwrite ( const void * ptr, size_t size, size_t count, FILE * stream );
Запись в файл
const void * ptr - указатель на массив
size_t size size - размер элемента
size_t count - количество элементов
FILE * stream - файл, в который будет записана информация
Возвращает количество удачно записанных символов

	size_t fread ( void * ptr, size_t size, size_t count, FILE * stream );
Чтение из файла
void * ptr - массив, куда будут записаны данные
size_t size - размер считываемых данных
size_t count - количество считываемых данных
FILE * stream - файл, из которого будет происходить считывание
Возвращает количество удачно считанных символов

	int fseek ( FILE * stream, long int offset, int origin );
Устанавливает каретку в файле на указанное место
При открытии файла позиция равна 0
FILE * stream - указатель на файл
long int offset - количество символов, на которое надо сдвинуть каретку
int origin - 1,2 или 3 (1 - Начало файла, 2 - Текущее положение, 3 - конец файла)
В случае удачной работы возвращает 0

	long int ftell ( FILE * stream );
Возвращает текущее положение каретки в файле FILE * stream
